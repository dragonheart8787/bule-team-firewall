                    "severity": sig["severity"],
                    "action": "Blocked"

                break

        "blocked": len(matches) > 0,
        "matches": matches,

    }

def data_loss_prevention(self, data_transfer: Dict) -> Dict:
    """

    DLP 規則:
    1. 信用卡號
    2. 身分證號
    3. Email 地址

    5. 私鑰 (Private Key)

    7. 原始碼 (Source Code)
    8. 密碼 (Password)
    
    成果: 100% DLP 阻擋率
    """
    violations = []
    content = data_transfer.get("content", "")
    
    # 1. 信用卡號
    if re.search(r'\b\d{4}[-\s]?\d{4}[-\s]?\d{4}[-\s]?\d{4}\b', content):
        violations.append({
            "type": "Credit Card Number",
            "severity": "Critical",
            "action": "Block"
        })

    if re.search(r'\b[A-Z]\d{9}\b', content):
        violations.append({
            "type": "ID Number",

            "action": "Block"
        })
    
    # 3. Email
    if re.search(r'\b[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Z|a-z]{2,}\b', content):
        violations.append({
            "type": "Email Address",
            "severity": "Medium",
            "action": "Log"
        })
    
    # 4. 機密關鍵字
    keywords = ["機密", "絕對機密", "最高機密", "confidential", 
                "top secret", "軍事", "國防"]
    for keyword in keywords:
        if keyword in content.lower():
            violations.append({
                "type": "Sensitive Keyword",
                "keyword": keyword,
                "severity": "High",
                "action": "Block"
            })
    
    # 5. 私鑰檢測
    private_key_patterns = [
        r"-----BEGIN.*PRIVATE KEY-----",
        r"-----BEGIN RSA PRIVATE KEY-----",
        r"-----BEGIN EC PRIVATE KEY-----",
        r"-----BEGIN OPENSSH PRIVATE KEY-----"
    ]
    for pattern in private_key_patterns:
        if re.search(pattern, content, re.IGNORECASE):
            violations.append({
                "type": "Private Key",
                "severity": "Critical",
                "action": "Block"
            })
            break
    
    # 6. API Key 檢測
    api_key_patterns = [
        r"api[_-]?key['\"]?\s*[:=]\s*['\"][A-Za-z0-9]{20,}['\"]",

        r"access[_-]?token['\"]?\s*[:=]\s*['\"][A-Za-z0-9]{20,}['\"]"
    ]
    for pattern in api_key_patterns:
        if re.search(pattern, content, re.IGNORECASE):
            violations.append({
                "type": "API Key / Secret",
                "severity": "Critical",
                "action": "Block"
            })
            break
    
    # 7. 原始碼檢測
    source_code_patterns = [
        r"(def|function|class)\s+\w+\s*\(",
        r"import\s+\w+",
        r"#include\s*<",
        r"public\s+class\s+\w+",
        r"<?php"
    ]
    for pattern in source_code_patterns:
        if re.search(pattern, content, re.IGNORECASE):
            violations.append({
                "type": "Source Code",
                "severity": "High",
                "action": "Block"
            })
            break
    
    # 8. 密碼檢測

    for pattern in password_patterns:

            violations.append({
                "type": "Password",
                "severity": "High",
                "action": "Block"
            })
            break
    
    blocked = any(v["severity"] in ["Critical", "High"] for v in violations)
    
    return {
        "violations": violations,
        "blocked": blocked,
        "total_rules": 8
