### 5. 加密算法正確性測試
```python
# 創建測試腳本: crypto_validation.py
from cryptography.hazmat.primitives.ciphers.aead import AESGCM
from cryptography.hazmat.primitives import hashes
from cryptography.hazmat.primitives.asymmetric import ec
import os

def test_aes_gcm():
    """測試 AES-256-GCM"""
    key = AESGCM.generate_key(bit_length=256)
    aesgcm = AESGCM(key)
    nonce = os.urandom(12)
    plaintext = b"Test data for CNSA 2.0 compliance"
    ciphertext = aesgcm.encrypt(nonce, plaintext, None)
    decrypted = aesgcm.decrypt(nonce, ciphertext, None)
    assert plaintext == decrypted
    print("✅ AES-256-GCM: PASS")

def test_ecdh_p384():
    """測試 ECDH P-384"""
    private_key = ec.generate_private_key(ec.SECP384R1())
    public_key = private_key.public_key()
    print("✅ ECDH P-384: PASS")

def test_sha384():
    """測試 SHA-384"""
    digest = hashes.Hash(hashes.SHA384())
    digest.update(b"Test data")
    hash_value = digest.finalize()
    assert len(hash_value) == 48  # 384 bits = 48 bytes
    print("✅ SHA-384: PASS")

if __name__ == "__main__":
    test_aes_gcm()
    test_ecdh_p384()
    test_sha384()
    print("\n🎖️ 所有 CNSA 2.0 加密測試通過!")
```

## 階段四：DDoS 防護測試 (1小時)

### 8. hping3 壓力測試
```bash
# 安裝 hping3
sudo apt install hping3

# 測試 SYN Flood 防護
sudo hping3 -S -p 80 --flood --rand-source 127.0.0.1

# 測試 UDP Flood 防護
sudo hping3 --udp -p 53 --flood --rand-source 127.0.0.1

# 測試 ICMP Flood 防護
sudo hping3 --icmp --flood 127.0.0.1

# 監控您的防火牆日誌,確認攻擊被檢測和阻斷
```

#!/bin/bash

      --profile moderate \

      --report oscap_report.html \
      /usr/share/xml/scap/ssg/content/ssg-ubuntu2004-ds.xml

echo "✅ 所有測試完成!"
echo "📁 報告位置: test_results/"
