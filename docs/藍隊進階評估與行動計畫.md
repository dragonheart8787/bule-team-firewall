```python
# 1. 自動化證據收集器
class EvidenceCollector:
    def collect_evidence(self, incident_id):
        evidence = {
            "id": incident_id,
            "timestamp": datetime.now(timezone.utc).isoformat(),
            "collector": "automated-system",
            "items": []
        }
        
        # 收集日誌
        logs = self._collect_logs(incident_id)
        evidence["items"].append({
            "type": "logs",
            "hash": hashlib.sha256(logs).hexdigest(),
            "size": len(logs),
            "location": f"/evidence/{incident_id}/logs.json"
        })
        
        # 收集 PCAP
        pcap = self._collect_pcap(incident_id)
        evidence["items"].append({
            "type": "pcap",
            "hash": hashlib.sha256(pcap).hexdigest(),
            "size": len(pcap),
            "location": f"/evidence/{incident_id}/capture.pcap"
        })
        
        # HSM 簽名
        signature = self._hsm_sign(json.dumps(evidence))
        evidence["signature"] = signature
        
        # 生成 manifest
        self._generate_manifest(evidence)
        
        return evidence
```

```python
# ATT&CK 映射引擎
class ATTACKMapper:
    def __init__(self):
        self.attack_matrix = self._load_attack_matrix()
        self.detection_rules = self._load_detection_rules()
    
    def generate_coverage_report(self):
        coverage = {}
        
        for technique in self.attack_matrix:
            technique_id = technique['id']  # e.g., T1059
            
            # 檢查偵測能力
            detection = self._check_detection(technique_id)
            
            # 檢查封鎖能力
            blocking = self._check_blocking(technique_id)
            
            # 檢查響應能力
            response = self._check_response(technique_id)
            
            # 收集證據
            evidence = self._collect_evidence(technique_id)
            
            coverage[technique_id] = {
                "name": technique['name'],
                "detection": detection,
                "blocking": blocking,
                "response": response,
                "evidence": evidence,
                "status": self._calculate_status(detection, blocking, response)
            }
        
        return self._generate_html_report(coverage)
    
    def _calculate_status(self, detection, blocking, response):
        if detection and blocking and response:
            return "FULL"  # 綠色
        elif detection and (blocking or response):
            return "PARTIAL"  # 黃色
        elif detection:
            return "DETECT_ONLY"  # 橙色
        else:
            return "NO_COVERAGE"  # 紅色
```

```python
# CTI 整合引擎
class CTIEngine:
    def ingest_ioc(self, ioc_source):
        """從 STIX/TAXII 源攝取 IoC"""
        iocs = self._fetch_from_taxii(ioc_source)
        
        for ioc in iocs:
            # 1. 儲存到威脅情報庫
            self._store_ioc(ioc)
            
            # 2. 自動更新檢測規則
            self._update_detection_rules(ioc)
            
            # 3. 測試是否會被偵測（validation）
            self._test_detection(ioc)
            
            # 4. 生成 IoC 報告
            self._generate_ioc_report(ioc)
    
    def _test_detection(self, ioc):
        """測試 IoC 是否會被系統偵測"""
        # 在隔離環境中模擬 IoC
        test_result = self._simulate_ioc(ioc)
        
        if not test_result['detected']:
            # False Negative - 需要改進規則
            self._alert_false_negative(ioc)
        
        return test_result
```
